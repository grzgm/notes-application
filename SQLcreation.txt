-- Note Table
CREATE TABLE dbo.notes
	(
	Id int NOT NULL PRIMARY KEY IDENTITY (1, 1) NOT FOR REPLICATION,
	UserId int NOT NULL,
	Title nchar(15) NULL,
	[Text] nchar(100) NULL,
	[Date] datetime2(0) NULL,
	EditDate datetime2(0) NULL
	);

-- HistoryNote Table
CREATE TABLE dbo.historyNotes
	(
	NoteId int NOT NULL,
	EditId int NOT NULL,
	UserId int NOT NULL,
	Title nchar(15) NULL,
	[Text] nchar(100) NULL,
	[Date] datetime2(0) NULL,
    CONSTRAINT PK_Person PRIMARY KEY (NoteId, EditId, UserId)
	);

CREATE TABLE dbo.account
(
	Id int NOT NULL PRIMARY KEY IDENTITY (1, 1) NOT FOR REPLICATION,
	[Name] nchar(15) NOT NULL,
	Email nchar(50) NULL,
	[Password] nchar(10) NULL,
	MaxAmountOfNotes int NULL,
	MaxLengthOfNote int NULL
);

CREATE TABLE dbo.userTable
(
	Id int NOT NULL FOREIGN KEY REFERENCES [account]([id]) NOT FOR REPLICATION,
	IsPremium bit NULL
);

CREATE TABLE dbo.premiumUserTable
(
	Id int NOT NULL FOREIGN KEY REFERENCES [account]([id]) NOT FOR REPLICATION,
	StartPremiumDate datetime2(0) NULL,
	EndPremiumDate datetime2(0) NULL
);

CREATE TABLE dbo.AdminTable
(
	Id int NOT NULL PRIMARY KEY IDENTITY (1, 1) NOT FOR REPLICATION,
);
